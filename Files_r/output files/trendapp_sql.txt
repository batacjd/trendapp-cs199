create table users (
	userid serial not null primary key,
	password character varying not null,
	email character varying not null
	);
	
create table superusers (
	superuserid serial not null primary key,
	userid integer not null references users (userid)
	);
	
create table userinfos (
	userinfoid serial not null primary key,
	userid integer not null references users (userid),
	givenname character varying not null,
	lastname character varying not null,
	birthmonth integer not null,
	birthdate integer not null,
	birthyear integer not null
	);
	
create table categories (
	categoryid serial not null primary key,
	categoryname character varying not null
	);
	
create table unittypes (
	unittypeid serial not null primary key,
	unittypename character varying not null
	);

create table units (
	unitid serial not null primary key,
	unitname character varying not null,
	categoryid integer not null references categories (categoryid),
	unittypeid integer not null references unittypes (unittypeid),
	coord_x numeric not null,
	coord_y numeric not null,
	startdate timestamp without time zone,
	enddate timestamp without time zone,
	rating double precision,
	mechanics character varying
	);

create table keywords (
	keywordid serial not null primary key,
	keyword character varying not null,
	categoryid integer not null references categories (categoryid)
	);

create table ratings (
	ratingid serial not null primary key,
	unitid integer not null references units (unitid),
	userid integer not null references users (userid),
	rating integer
	);

create table months (
	monthid serial not null primary key,
	month character varying not null
	);
